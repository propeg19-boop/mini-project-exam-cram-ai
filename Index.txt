    <script>
        // State
        let currentSelectedTopic = "";

        // Elements
        const form = document.getElementById('study-form');
        const planSection = document.getElementById('plan-section');
        const answerSection = document.getElementById('answer-section');
        const planTableBody = document.getElementById('plan-table-body');
        const dayPlanContainer = document.getElementById('day-plan-container');
        const modal = document.getElementById('study-modal');
        const submitBtn = document.getElementById('generate-btn');

        // Form Submit Handler
        form.addEventListener('submit', async (e) => {
            e.preventDefault();
            
            const topics = document.getElementById('topics').value;
            const days = document.getElementById('days').value;
            const skill = document.getElementById('skill').value;
            const tone = document.getElementById('tone').value;

            submitBtn.disabled = true;
            submitBtn.textContent = 'Analyzing...';
            document.getElementById('plan-days-badge').textContent = days;

            const payload = {
                questions: topics,
                days: parseInt(days, 10),
                level: skill,
                tone: tone
            };

            try {
                const response = await fetch('/generate-plan', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });

                if (!response.ok) throw new Error('Backend not available. Using dummy data.');
                const data = await response.json();
                
                renderPlanTable(data.table);
                
                // 🟢 FIX 4: Safety check for day_plan
                if (data.day_plan) {
                    renderDayPlan(data.day_plan);
                }

            } catch (error) {
                console.warn(error.message);
                
                // 🟢 FIX 2: Fixed Dummy Data keys to match backend schema perfectly
                const dummyResponse = {
                    table: [
                        { priority: "High", question: "Understanding Core Concepts", estimated_time: "2 Hours" },
                        { priority: "Medium", question: "Practical Implementation", estimated_time: "1.5 Hours" },
                        { priority: "Low", question: "Historical Context & Trivia", estimated_time: "30 Mins" }
                    ],
                    day_plan: {
                        "Day 1": ["Understanding Core Concepts"],
                        "Day 2": ["Practical Implementation", "Historical Context & Trivia"]
                    }
                };
                
                setTimeout(() => {
                    renderPlanTable(dummyResponse.table);
                    if (dummyResponse.day_plan) {
                        renderDayPlan(dummyResponse.day_plan);
                    }
                }, 800);
            } finally {
                submitBtn.disabled = false;
                submitBtn.textContent = 'Generate Study Plan';
            }
        });

        // Render Day Plan
        function renderDayPlan(dayPlanObj) {
            dayPlanContainer.innerHTML = '';
            
            Object.entries(dayPlanObj).forEach(([day, topics]) => {
                const card = document.createElement('div');
                card.className = "bg-indigo-50/50 border border-indigo-100 rounded-lg p-4";
                
                const topicsList = topics.map(t => `<li class="truncate text-sm text-gray-700" title="${t}">• ${t}</li>`).join('');
                
                card.innerHTML = `
                    <h4 class="font-bold text-indigo-700 mb-2">${day}</h4>
                    <ul class="space-y-1">
                        ${topicsList}
                    </ul>
                `;
                dayPlanContainer.appendChild(card);
            });
        }

        // Render Table
        function renderPlanTable(planArray) {
            planTableBody.innerHTML = '';
            
            // 🟢 FIX 1: Updated to row.question and row.estimated_time
            planArray.forEach(row => {
                const tr = document.createElement('tr');
                tr.className = "hover:bg-indigo-50 cursor-pointer transition-colors duration-150 group";
                tr.onclick = () => openModal(row.question);

                let badgeColor = 'bg-gray-100 text-gray-800';
                if (row.priority === 'High') badgeColor = 'bg-red-100 text-red-800';
                if (row.priority === 'Medium') badgeColor = 'bg-yellow-100 text-yellow-800';
                if (row.priority === 'Low') badgeColor = 'bg-green-100 text-green-800';

                tr.innerHTML = `
                    <td class="px-6 py-4 whitespace-nowrap">
                        <span class="px-2 inline-flex text-xs leading-5 font-semibold rounded-full ${badgeColor}">${row.priority}</span>
                    </td>
                    <td class="px-6 py-4 text-sm font-medium text-gray-900 group-hover:text-indigo-600">${row.question}</td>
                    <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500">${row.estimated_time}</td>
                    <td class="px-6 py-4 text-sm text-gray-500 max-w-xs truncate">Keyword-based priority</td>
                `;
                planTableBody.appendChild(tr);
            });

            planSection.classList.remove('hidden');
            planSection.scrollIntoView({ behavior: 'smooth', block: 'start' });
        }

        // Modal Controls
        function openModal(topic) {
            currentSelectedTopic = topic;
            document.getElementById('modal-topic-name').textContent = `Topic: ${topic}`;
            modal.classList.remove('hidden');
        }

        function closeModal() {
            modal.classList.add('hidden');
            currentSelectedTopic = "";
        }

        // Fetch Answer & Diagrams
        async function fetchAnswer(mode) {
            closeModal();
            const tone = document.getElementById('tone').value;
            const skillLevel = document.getElementById('skill').value;
            
            const warningBox = document.getElementById('quick-warning');
            warningBox.style.display = mode === 'quick' ? 'block' : 'none';

            document.getElementById('answer-topic-title').textContent = "Generating Answer...";
            answerSection.classList.remove('hidden');
            answerSection.scrollIntoView({ behavior: 'smooth', block: 'start' });

            const payload = {
                question: currentSelectedTopic,
                mode: mode === 'focus' ? 'focused' : 'quick',
                level: skillLevel,
                tone: tone
            };

            try {
                const ansResponse = await fetch('/generate-answer', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });
                
                const imgResponse = await fetch('/get-images', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ topic: currentSelectedTopic })
                });

                if (!ansResponse.ok || !imgResponse.ok) throw new Error('API missing. Using dummy data.');
                
                const ansData = await ansResponse.json();
                const imgData = await imgResponse.json();
                
                renderAnswer(currentSelectedTopic, mode, ansData, imgData.images);

            } catch (error) {
                console.warn(error.message);
                
                const isQuick = mode === 'quick';
                const dummyAnswer = {
                    analogy: `Think of "${currentSelectedTopic}" like a recipe. You need all the ingredients (concepts) in the right order to bake the cake.`,
                    understanding: isQuick ? `Core formula/definition of ${currentSelectedTopic} in 2 lines.` : `Detailed step-by-step breakdown of ${currentSelectedTopic} covering all edge cases.`,
                    answer: `1. Definition: Clearly state what it is.<br>2. Mechanism: Explain how it works.<br>3. Advantages: Point A, Point B.<br>4. Disadvantages: Point X, Point Y.<br><i>Conclusion summarizing the topic.</i>`,
                    extra: "Mnemonic: 'C-A-R-E' (Context, Action, Result, Evaluation)"
                };

                const dummyImages = [
                    `https://placehold.co/400x300/e2e8f0/4f46e5?text=Diagram+1\\n${encodeURIComponent(currentSelectedTopic)}`,
                    `https://placehold.co/400x300/e2e8f0/4f46e5?text=Diagram+2\\nFlowchart`,
                    `https://placehold.co/400x300/e2e8f0/4f46e5?text=Diagram+3\\nComparison`
                ];

                setTimeout(() => renderAnswer(currentSelectedTopic, mode, dummyAnswer, dummyImages), 1000);
            }
        }

        // Render Answer UI
        function renderAnswer(topic, mode, data, images) {
            document.getElementById('answer-topic-title').textContent = topic;
            
            const badge = document.getElementById('answer-mode-badge');
            badge.textContent = mode === 'quick' ? '⚡ Quick Revision' : '🧠 Focus Mode';
            badge.className = mode === 'quick' 
                ? 'bg-yellow-100 text-yellow-800 text-xs font-semibold px-2.5 py-1 rounded-md' 
                : 'bg-indigo-100 text-indigo-800 text-xs font-semibold px-2.5 py-1 rounded-md';

            document.getElementById('ans-analogy').innerHTML = data.analogy || '';
            document.getElementById('ans-quick').innerHTML = data.understanding || '';
            document.getElementById('ans-exam').innerHTML = data.answer || '';
            document.getElementById('ans-optional').innerHTML = data.extra || '';

            // Render Images 
            const grid = document.getElementById('diagram-grid');
            grid.innerHTML = '';
            
            // 🟢 FIX 3: Safety check for undefined images array
            (images || []).forEach((url, idx) => {
                const imgContainer = document.createElement('div');
                imgContainer.className = "rounded-lg overflow-hidden border border-gray-200 shadow-sm bg-gray-50 flex items-center justify-center";
                imgContainer.innerHTML = `<img src="${url}" alt="Exam Diagram ${idx+1}" class="w-full h-auto object-cover hover:scale-105 transition-transform duration-300">`;
                grid.appendChild(imgContainer);
            });
            
            setTimeout(() => {
                answerSection.scrollIntoView({ behavior: 'smooth', block: 'start' });
            }, 100);
        }
    </script>
